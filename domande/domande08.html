			case 1:
				System.out.print ("Uno");
				break;
			default:
				System.out.println ("Errore");
		} 
	}

		for (int i=0; i<=2; i++)
			conv (i);
	}
}

  • B L'esecuzione del metodo main causa la stampa di ZeroUnoUnoErrore

  • D L'esecuzione del metodo main causa la stampa di ZeroUno
